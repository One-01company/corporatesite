                renderer.setSize(w, h);

                renderer.setPixelRatio(Math.min(window.devicePixelRatio, 2));

                mountRef.current.appendChild(renderer.domElement);


                const group = new THREE.Group();

                scene.add(group);


                // --- Network Effect Configuration ---

                const particleCount = 100; // Number of dots (Optimized)

                const r = 120; // Spread radius

                const rHalf = r / 2;

                const connectionDist = 25; // Distance to connect dots


                const particlesData = [];

                const positions = new Float32Array(particleCount * 3);

                const linePositions = new Float32Array(particleCount * particleCount * 3);


                // 1. Initialize Particles

                const pMaterial = new THREE.PointsMaterial({

                    color: 0x999999,

                    size: 0.6,

                    transparent: true,

                    opacity: 0.6,

                    sizeAttenuation: true

                });


                for (let i = 0; i < particleCount; i++) {

                    const x = Math.random() * r - rHalf;

                    const y = Math.random() * r - rHalf;

                    const z = Math.random() * r - rHalf;


                    positions[i * 3] = x;

                    positions[i * 3 + 1] = y;

                    positions[i * 3 + 2] = z;


                    particlesData.push({

                        velocity: new THREE.Vector3(

                            -0.05 + Math.random() * 0.1,

                            -0.05 + Math.random() * 0.1,

                }

                const particleSystem = new THREE.Points(particlesGeo, pMaterial);

                group.add(particleSystem);


                // 2. Initialize Lines

                linesGeo.setAttribute('position', new THREE.BufferAttribute(linePositions, 3).setUsage(THREE.DynamicDrawUsage));

                // --- Animation Loop ---

                let animationId;

                

                const animate = () => {

                    let vertexpos = 0;

                    let numConnected = 0;


                    // Update Particles

                    for (let i = 0; i < particleCount; i++) {

                        const p = particlesData[i];

                        

                        // Update position

                        positions[i * 3]     += p.velocity.x;

                        positions[i * 3 + 1] += p.velocity.y;

                        positions[i * 3 + 2] += p.velocity.z;


                        // Bounce off boundary

                        if (positions[i * 3] < -rHalf || positions[i * 3] > rHalf) p.velocity.x = -p.velocity.x;

                        if (positions[i * 3 + 1] < -rHalf || positions[i * 3 + 1] > rHalf) p.velocity.y = -p.velocity.y;

                        if (positions[i * 3 + 2] < -rHalf || positions[i * 3 + 2] > rHalf) p.velocity.z = -p.velocity.z;


                        // Check Connections

                        for (let j = i + 1; j < particleCount; j++) {

                            const dx = positions[i * 3] - positions[j * 3];

                            const dy = positions[i * 3 + 1] - positions[j * 3 + 1];

                            const dz = positions[i * 3 + 2] - positions[j * 3 + 2];

                            const dist = Math.sqrt(dx * dx + dy * dy + dz * dz);


                            if (dist < connectionDist) {

                                // Add line vertices

                                linePositions[vertexpos++] = positions[i * 3];

                                linePositions[vertexpos++] = positions[i * 3 + 1];

                                linePositions[vertexpos++] = positions[i * 3 + 2];


                                linePositions[vertexpos++] = positions[j * 3];

                                linePositions[vertexpos++] = positions[j * 3 + 1];

                                linePositions[vertexpos++] = positions[j * 3 + 2];


                                numConnected++;

                            }

                        }

                    }

                    particleSystem.geometry.attributes.position.needsUpdate = true;

                    group.rotation.x += 0.0001;
